% Resume Item
\newcommand{\resumeItem}[4]{
    \begin{tabular*}{0.99\textwidth}[t]{l@{\extracolsep{\fill}}r}
      \textbf{#1} & #2 \\
      \textit{\small#3} & \textit{\small #4} \\
    \end{tabular*}%\vspace{5pt}
}
% Resume Item Partial (position heading)
\newcommand{\resumeItemPart}[2]{
    \begin{tabular*}{0.99\textwidth}[t]{l@{\extracolsep{\fill}}r}
      \textit{\small#1} & \textit{\small #2} \\
    \end{tabular*}%\vspace{-5pt}
}
% Resume Item Minimal
\newcommand{\resumeItemMin}[2]{
    \begin{tabular*}{0.99\textwidth}[t]{l@{\extracolsep{\fill}}r}
      \textbf{#1} & #2 \\
    \end{tabular*}\vspace{5pt}
}

% Bullet List Environment
\newcommand{\resumeBulletStart}{\vspace{-5pt}\begin{itemize}}%[leftmargin=*]}
\newcommand{\resumeBulletEnd}{\end{itemize}}
% Bullet
\newcommand{\resumeBullet}[2][]{%
  \item\small{%
      \ifblank{#1}{}{\textbf{#1: }}{#2}
      \vspace{-4pt}% Output #2
  }
}

% List Enviornment
\newcommand{\resumeListStart}{\begin{itemize}[leftmargin=*]}
\newcommand{\resumeListEnd}{\end{itemize}\vspace{-5pt}}
% List Item
\newcommand{\resumeListItem}[2][]{%
    \resumeBullet[#1]{#2}
    \vspace{-2pt}
}

%  ────────────────────────────────────────────────────────────────────
\section{Projects}

  $for(projects)$
    \resumeItem{$it.project$}{}
    {$it.skills$}{$it.dates$}
    \resumeBulletStart
      $for(it.description)$
          \resumeBullet[$if(it.keyword)$$it.keyword$$endif$]{$it.text$}
      $endfor$
    \resumeBulletEnd
  $endfor$

 % ────────────────────────────────────────────────────────────────────
%   EXPERIENCE                                                         
%  ────────────────────────────────────────────────────────────────────
\section{Experience}

  $for(experience)$
    $if(it.company)$
      \resumeItem
        {$it.company$}{$it.location$}
        {$it.position$}{$it.dates$}
        \resumeBulletStart
          $for(it.description)$
              \resumeBullet[$if(it.keyword)$$it.keyword$$endif$]{$it.text$}
          $endfor$
        \resumeBulletEnd
    $else$
      \resumeItemPart
        {$it.position$}{$it.dates$}
        \resumeBulletStart
          $for(it.description)$
              \resumeBullet[$if(it.keyword)$$it.keyword$$endif$]{$it.text$}
          $endfor$
        \resumeBulletEnd
    $endif$
  $endfor$

%  ────────────────────────────────────────────────────────────────────
